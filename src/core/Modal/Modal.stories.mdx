# Modal

Displays a Modal

export const Template = ({
    title,
    description,
    borderColor,
    shadowColor,
    titleSize,
    titleColor,
    descriptionSize,
    descriptionColor,
    ...args
}) => {
    return (
        <Modal
            title={title}
            description={description}
            shadowColor={shadowColor}
            borderColor={borderColor}
            titleSize={titleSize}
            descriptionSize={descriptionSize}
            titleColor={titleColor}
            descriptionColor={descriptionColor}
            {...args}
        />
    );
};
