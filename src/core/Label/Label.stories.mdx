# Label

Displays a Label

export const Template = ({ labelColor, required, showPointer, labelSize, labelWeight, lineHeight, upperCase, ...args }) => {
    return (
        <Label
            labelColor={labelColor}
            required={required}
            showPointer={showPointer}
            labelSize={labelSize}
            labelWeight={labelWeight}
            lineHeight={lineHeight}
            uppercase={upperCase}
            {...args}
        >
            Lorem Ipsum is simply dummy text
        </Label>
    );
};
